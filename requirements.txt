#       pip download torch==2.6.0 torchvision==0.21.0 --index-url https://download.pytorch.org/whl/cpu -d ./offline_packages
#
# [On offline local machine]
# 5. Install Python 3.11 (same version)
# 6. Create venv:
#       py -3.11 -m venv venv
#       venv\Scripts\activate
# 7. Install all packages offline:
#       pip install --no-index --find-links=./offline_packages -r requirements.txt
